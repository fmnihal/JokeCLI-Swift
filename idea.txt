import Foundation
struct JokeFetcher {
    static func fetch(completion: @escaping (String) -> Void) {
        let url = URL(string: "https://official-joke-api.appspot.com/random_joke")!
        let task = URLSession.shared.dataTask(with: url) { data, _, error in
            guard let data = data, error == nil else {
                completion("Failed to fetch joke.")
                return
            }
            if let joke = try? JSONDecoder().decode(Joke.self, from: data) {
                completion("\(joke.setup)\n\(joke.punchline)")
            } else {
                completion("Couldn't parse the joke.")
            }
        }
        task.resume()
    }
}
struct Joke: Decodable {
    let setup: String
    let punchline: String
}
